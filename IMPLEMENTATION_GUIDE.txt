2️⃣ BLOG SYSTEM IMPLEMENTATION
──────────────────────────────

Create blog.html:

```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Blog - Shahil Sinha</title>
    <link rel="stylesheet" href="advanced-styles.css">
    <style>
        .blog-container { max-width: 1000px; margin: 0 auto; padding: 2rem; }
        .blog-header { text-align: center; margin-bottom: 3rem; }
        .blog-grid { display: grid; gap: 2rem; }
        .blog-card {
            background: rgba(0, 212, 255, 0.05);
            border: 1px solid rgba(0, 212, 255, 0.2);
            padding: 1.5rem;
            border-radius: 10px;
        }
        .blog-card h3 { color: #00d4ff; }
        .blog-meta { color: #a0aec0; font-size: 0.9rem; margin: 0.5rem 0; }
    </style>
</head>
<body>
    <div class="blog-container">
        <div class="blog-header">
            <h1>My Blog</h1>
            <p>Thoughts on coding, AI, and tech</p>
        </div>
        <div class="blog-grid" id="blog-list"></div>
    </div>

    <script src="firebase-config.js"></script>
    <script>
        if (typeof firebase !== 'undefined') {
            const db = firebase.firestore();
            
            db.collection('blog_posts')
              .orderBy('date', 'desc')
              .get()
              .then(snapshot => {
                  const blogList = document.getElementById('blog-list');
                  
                  if (snapshot.empty) {
                      blogList.innerHTML = '<p>No posts yet</p>';
                      return;
                  }
                  
                  snapshot.forEach(doc => {
                      const post = doc.data();
                      const date = new Date(post.date).toLocaleDateString();
                      
                      const article = `
                          <article class="blog-card">
                              <h3>${post.title}</h3>
                              <div class="blog-meta">
                                  ${date} • ${post.category}
                              </div>
                              <p>${post.excerpt}</p>
                              <a href="blog-post.html?id=${doc.id}">Read More →</a>
                          </article>
                      `;
                      
                      blogList.innerHTML += article;
                  });
              });
        }
    </script>
</body>
</html>
```

function trackPageView(page) {
    const db = firebase.firestore();
    db.collection('analytics').add({
        page: page,
        timestamp: new Date(),
        userAgent: navigator.userAgent,
        referrer: document.referrer
    });
}

// Track on page load
trackPageView(window.location.pathname);

Step 3: Create Service Worker

Create file: service-worker.js

```javascript
const CACHE_NAME = 'shahil-portfolio-v1';
const urlsToCache = [
  '/',
  '/index.html',
  '/admin.html',
  '/blog.html',
  '/advanced-styles.css',
  '/firebase-config.js',
  '/assets/Rezume__1_.pdf'
];

self.addEventListener('install', event => {
  event.waitUntil(
    caches.open(CACHE_NAME)
      .then(cache => cache.addAll(urlsToCache))
  );
});

self.addEventListener('fetch', event => {
  event.respondWith(
    caches.match(event.request)
      .then(response => response || fetch(event.request))
      .catch(() => caches.match('/index.html'))
  );
});
```


Step 4: Register Service Worker

In portfolio.html, before </body>:

```html
<script>

            .then(registration => {
                console.log('SW registered', registration);
            });
    }
</script>
```

7️⃣ SECURITY HARDENING
──────────────────────

Step 1: Update Firebase Security Rules

In Firestore:

```javascript
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow reading blog posts publicly
    match /blog_posts/{document=**} {
      allow read: if true;
    }
    
    // Allow creating contact forms but not reading
    match /contacts/{document=**} {
      allow create: if request.data.email != null
                    && request.data.message != null;
    }
    
    // Deny all other access
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
```
